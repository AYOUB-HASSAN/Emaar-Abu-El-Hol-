const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID_HERE';
const DRIVE_FOLDER_ID = 'YOUR_DRIVE_FOLDER_ID_HERE';

function doGet(e) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  
  // Fetch Stats Data
  const dataSheet = ss.getSheetByName('Data');
  const dataValues = dataSheet.getDataRange().getValues();
  const stats = {};
  dataValues.slice(1).forEach(row => {
    stats[row[0]] = isNaN(row[1]) ? row[1] : Number(row[1]);
  });

  // Fetch Gallery Data
  const gallerySheet = ss.getSheetByName('Gallery');
  const galleryValues = gallerySheet.getDataRange().getValues();
  const gallery = galleryValues.slice(1).map(row => ({
    id: row[0],
    type: row[1],
    url: row[2],
    category: row[3],
    date: row[4],
    status: row[5]
  }));

  // Fetch Leaderboard (Calculated or from a Sheet)
  // For simplicity, let's assume a "Leaderboard" sheet exists
  const leaderSheet = ss.getSheetByName('Leaderboard');
  const leaders = leaderSheet.getDataRange().getValues().slice(1).map(row => ({
    batch: row[0],
    amount: Number(row[1]),
    label: row[2]
  }));

  const payload = {
    stats: stats,
    gallery: gallery,
    leaders: leaders
  };

  return ContentService.createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const folder = DriveApp.getFolderById(DRIVE_FOLDER_ID);
  
  try {
    const data = JSON.parse(e.postData.contents);
    
    if (data.action === 'analyze') {
      const bytes = Utilities.base64Decode(data.file.split(",")[1]);
      const blob = Utilities.newBlob(bytes, "image/jpeg", "temp_ocr.jpg");
      
      // OCR trick: Create a Google Doc from image
      const resource = { title: "temp_ocr", mimeType: "image/jpeg" };
      const tempDoc = Drive.Files.insert(resource, blob, { ocr: true });
      
      // Get text from the doc
      const doc = DocumentApp.openById(tempDoc.id);
      const text = doc.getBody().getText();
      
      // Cleanup
      Drive.Files.remove(tempDoc.id);

      // Simple extraction: find largest number that looks like an amount (3+ digits)
      const amounts = text.match(/\d{3,}/g) || [];
      const suggested = amounts.length > 0 ? Math.max(...amounts.map(Number)) : 0;
      
      return ContentService.createTextOutput(JSON.stringify({ status: 'success', text: text, suggested: suggested }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (data.action === 'upload') {
      // Logic for file upload
      const contentType = data.file.split(";")[0].split(":")[1];
      const bytes = Utilities.base64Decode(data.file.split(",")[1]);
      const blob = Utilities.newBlob(bytes, contentType, data.filename);
      const file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      
      const fileUrl = 'https://lh3.googleusercontent.com/d/' + file.getId();
      
      // Log to Gallery Sheet
      const gallerySheet = ss.getSheetByName('Gallery');
      gallerySheet.appendRow([
        Utilities.getUuid(),
        'image',
        fileUrl,
        data.category,
        new Date().toISOString().split('T')[0],
        'active'
      ]);
      
      return ContentService.createTextOutput(JSON.stringify({ status: 'success', url: fileUrl }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    if (data.action === 'updateStats') {
      const dataSheet = ss.getSheetByName('Data');
      const rows = dataSheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.key) {
          dataSheet.getRange(i + 1, 2).setValue(data.value);
          break;
        }
      }
      return ContentService.createTextOutput(JSON.stringify({ status: 'success' }))
        .setMimeType(ContentService.MimeType.JSON);
    }

  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
